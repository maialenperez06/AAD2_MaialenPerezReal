CREACIÓN TABLA PELÍCULA

CREATE TABLE PELICULA ( 
    ID INT PRIMARY KEY IDENTITY(1,1), //para poder mostrar la info de la película más fácil
    Titulo VARCHAR(100),
    Estreno INT,
    Duracion INT,
    Precio Decimal(10,2), // para poder acceder a este atributo cuando muestra la info de la película
    Sinopsis TEXT,
    URL_Imagen VARCHAR(255),
    Entradas INT
);
